USING PL/SQL:
------------------------------------------------------------

printing the team with maximum goals.....:


SET SERVEROUTPUT ON;

CREATE OR REPLACE PROCEDURE find_most_scoring_team1 AS
    v_teamid NUMBER;
    v_teamname teams.teamname%TYPE;
    v_max_goals NUMBER := 0;
BEGIN
    FOR team_rec IN (SELECT t.teamid, t.teamname
                     FROM teams t
                          JOIN players p ON t.teamid = p.teamid
                          JOIN goals g ON p.playerid = g.playerid) LOOP
        -- Calculate total goals for each team
        DECLARE
            v_total_goals NUMBER;
        BEGIN
            SELECT COUNT(*)
            INTO v_total_goals
            FROM goals g
            JOIN players p ON g.playerid = p.playerid
            WHERE p.teamid = team_rec.teamid;

            -- Check if the total goals are greater than the current maximum
            IF v_total_goals > v_max_goals THEN
                v_max_goals := v_total_goals;
                v_teamid := team_rec.teamid;
                v_teamname := team_rec.teamname;
            END IF;
        END;

    END LOOP;

    -- Print the result
    DBMS_OUTPUT.PUT_LINE('Most Scoring Team: ' || v_teamname || ', Total Goals: ' || v_max_goals);
END;
/

-- Example usage
BEGIN
    find_most_scoring_team1;
END;
/






changing a random match date using PL/SQL.........:


SET SERVEROUTPUT ON;
CREATE OR REPLACE PROCEDURE update_match_date(
    p_match_id IN NUMBER,
    p_new_match_date IN DATE
) AS
BEGIN
    -- Update the match date
    UPDATE matches
    SET matchdate = p_new_match_date
    WHERE matchid = p_match_id;

    -- Commit the changes
    COMMIT;

    -- Display success message
    DBMS_OUTPUT.PUT_LINE('Match date updated successfully for Match ID ' || p_match_id);
EXCEPTION
    WHEN OTHERS THEN
        -- Display error message
        DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
END;
/

-- Example usage
BEGIN
    update_match_date(1, TO_DATE('25-NOV-23', 'DD-MON-YY'));
END;
/
